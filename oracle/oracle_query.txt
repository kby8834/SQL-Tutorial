-- 계층형 쿼리
WITH TEMP(PARENT_NAME, NAME) AS (
    SELECT NULL, '이강애' FROM DUAL UNION
    SELECT '이강애', '김보영' FROM DUAL UNION
    SELECT '이강애', '김성복' FROM DUAL UNION
    SELECT '김보영', '김조은' FROM DUAL UNION
    SELECT '김보영', '김평화' FROM DUAL UNION
    SELECT '김보영', '김사랑' FROM DUAL
)
SELECT
    LEVEL,
    NAME,
    LPAD(' ', (LEVEL - 1) * 2, ' ') || NAME
FROM TEMP
START WITH PARENT_NAME IS NULL
CONNECT BY PRIOR NAME = PARENT_NAME;
